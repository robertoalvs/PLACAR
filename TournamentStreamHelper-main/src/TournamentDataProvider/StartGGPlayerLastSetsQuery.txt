query PlayerLastSetsQuery($eventSlug: String!, $playerID: ID!) {
	event(slug: $eventSlug) {
    sets(page: 1, perPage: 10, sortType: RECENT, filters: {playerIds: [$playerID]}) {
      nodes {
        id
        slots {
            entrant {
                id
                participants {
                    id
                    player {
                        id
                        prefix
                        gamerTag
                    }
                }
            }
        }
        entrant1Score
        entrant2Score
        winnerId
        fullRoundText
        phaseGroup {
            displayIdentifier
            phase {
                name
            }
        }
      }
    }
  }
}
